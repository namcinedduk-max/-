🛠️ Google Apps Script 설정 방법

1
Google 스프레드시트 새로 만들기
sheets.google.com 에서 새 시트를 만드세요. 이름은 자유롭게.
2
확장 프로그램 → Apps Script 열기
상단 메뉴에서 확장 프로그램 → Apps Script 클릭
3
아래 코드를 Code.gs에 붙여넣기 (기존 내용 전체 삭제 후)
const SHEET_NAME = "주문장부"; const FUNERAL_SHEET_NAME = "장례식장장부"; function getOrCreateSheet(name, headers) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.getRange(1,1,1,headers.length).setValues([headers]); sheet.getRange(1,1,1,headers.length).setFontWeight("bold").setBackground("#3D2314").setFontColor("#FAF6EF"); } return sheet; } function doPost(e) { try { var params = JSON.parse(e.postData.contents); var isFuneral = params.sheet === "funeral"; if (isFuneral) { var sheet = getOrCreateSheet(FUNERAL_SHEET_NAME, ["id","접수시각","장례식장","빈소위치","떡종류","배송일","배송시간","상품금액","배송비","할인","총금액","결제상태","담당직원"]); if (params.action === "save") { var d = params.data; sheet.appendRow([d.id,d.createdAt,d.funeralName,d.binsoLocation,d.tteokType,d.deliveryDate,d.deliveryTime,d.priceProduct||0,d.priceDelivery||0,d.priceDiscount||0,d.priceTotal||0,d.payStatus,d.staffName]); } else if (params.action === "delete") { deleteRow(sheet, params.id); } } else { var sheet = getOrCreateSheet(SHEET_NAME, ["id","접수시각","배송일","배송시간","받는분","받는분 전화","보내는분","보내는분 전화","떡종류","배송지","상품금액","배송비","할인","총금액","결제방법","결제상태","담당직원"]); if (params.action === "save") { var d = params.data; sheet.appendRow([d.id,d.createdAt,d.deliveryDate,d.deliveryTime,d.receiverName,d.receiverPhone,d.senderName,d.senderPhone,d.tteokType,d.deliveryAddr1,d.priceProduct||0,d.priceDelivery||0,d.priceDiscount||0,d.priceTotal||0,d.payMethod,d.payStatus,d.staffName]); } else if (params.action === "delete") { deleteRow(sheet, params.id); } } return jsonOut({success:true}); } catch(err) { return jsonOut({success:false,error:err.message}); } } function doGet(e) { try { var isFuneral = e.parameter.sheet === "funeral"; if (e.parameter.action === "getOrders") { var sheet, cols; if (isFuneral) { sheet = getOrCreateSheet(FUNERAL_SHEET_NAME,["id","접수시각","장례식장","빈소위치","떡종류","배송일","배송시간","상품금액","배송비","할인","총금액","결제상태","담당직원"]); cols = 13; } else { sheet = getOrCreateSheet(SHEET_NAME,["id","접수시각","배송일","배송시간","받는분","받는분 전화","보내는분","보내는분 전화","떡종류","배송지","상품금액","배송비","할인","총금액","결제방법","결제상태","담당직원"]); cols = 17; } var last = sheet.getLastRow(); if (last < 2) return jsonOut({success:true,orders:[]}); var rows = sheet.getRange(2,1,last-1,cols).getValues(); var orders = rows.map(function(r){ if(isFuneral) return {id:r[0],createdAt:r[1],funeralName:r[2],binsoLocation:r[3],tteokType:r[4],deliveryDate:r[5],deliveryTime:r[6],priceProduct:r[7],priceDelivery:r[8],priceDiscount:r[9],priceTotal:r[10],payStatus:r[11],staffName:r[12]}; return {id:r[0],createdAt:r[1],deliveryDate:r[2],deliveryTime:r[3],receiverName:r[4],receiverPhone:r[5],senderName:r[6],senderPhone:r[7],tteokType:r[8],deliveryAddr1:r[9],deliveryAddr2:r[10]||'',labelText:r[11]||'',memo:r[12],priceProduct:r[13],priceDelivery:r[14],priceDiscount:r[15],priceTotal:r[16],payMethod:r[17],payStatus:r[18],staffName:r[19],deliveryMethod:r[20]||'배송',linkCode:r[21]||''}; }).reverse(); return jsonOut({success:true,orders:orders}); } return jsonOut({status:"ok",message:"연결 성공!"}); } catch(err) { return jsonOut({success:false,error:err.message}); } } function deleteRow(sheet, id) { var last = sheet.getLastRow(); for (var i=2;i<=last;i++) { if (String(sheet.getRange(i,1).getValue())===String(id)) { sheet.deleteRow(i); return; } } } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
